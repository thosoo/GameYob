76{
77 //---------------------------------------------------------------------------------
78 // clear sound registers
79 dmaFillWords(0, (void *)0x04000400, 0x100);
80
81 REG_SOUNDCNT |= SOUND_ENABLE;
82 writePowerManagement(PM_CONTROL_REG, (readPowerManagement(PM_CONTROL_REG) & ~PM_SOUND_MUTE) | PM_SOUND_AMP);
83 powerOn(POWER_SOUND);
84
85 readUserSettings();
86 ledBlink(0);
87
88 irqInit();
89 // Start the RTC tracking IRQ
90 // initClockIRQ(); //windwakr: Doesn't seem to work on 3DS.
91 resyncClock();
92
93 fifoInit();
94
95 while (!fifoCheckValue32(FIFO_USER_03))
96 ;
97 sharedData = (SharedData *)(fifoGetValue32(FIFO_USER_03) | 0x02000000);
98
99 SetYtrigger(80);
100
101 installWifiFIFO();
102 installSoundFIFO();
103
104 installSystemFIFO();
105
106 irqSet(IRQ_VCOUNT, VcountHandler);
107 irqSet(IRQ_VBLANK, VblankHandler);
108
109 irqEnable(IRQ_VBLANK | IRQ_VCOUNT | IRQ_NETWORK);
110
111 setPowerButtonCB(powerButtonCB);
112
113 installGameboySoundFIFO();
114
115 while (!exitflag)
116 {
117 if (0 == (REG_KEYINPUT & (KEY_SELECT | KEY_START | KEY_L | KEY_R)))
118 {
119 exitflag = true;
120 }
121 swiWaitForVBlank();
122 }
123 return 0;
