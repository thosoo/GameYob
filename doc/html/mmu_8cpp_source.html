123void handleHuC3Command(u8 cmd)
124{
125 switch (cmd & 0xf0)
126 {
127 case 0x10: /* Read clock */
128 if (HuC3Shift > 24)
129 break;
130
131 switch (HuC3Shift)
132 {
133 case 0:
134 case 4:
135 case 8: /* Minutes */
136 HuC3Value = (gbClock.huc3.m >> HuC3Shift) & 0xf;
137 break;
138 case 12:
139 case 16:
140 case 20: /* Days */
141 HuC3Value = (gbClock.huc3.d >> (HuC3Shift - 12)) & 0xf;
142 break;
143 case 24: /* Year */
144 HuC3Value = gbClock.huc3.y & 0xf;
145 break;
146 }
147 HuC3Shift += 4;
148 break;
149 case 0x40:
150 switch (cmd & 0xf)
151 {
152 case 0:
153 case 4:
154 case 7:
155 HuC3Shift = 0;
156 break;
157 }
158
159 latchClock();
160 break;
161 case 0x50:
162 break;
163 case 0x60:
164 HuC3Value = 1;
165 break;
166 default:
167 printLog("undandled HuC3 cmd %02x\n", cmd);
168 }
169}

298/* MBC3 */
299void m3w(u16 addr, u8 val)
300{
301 switch (addr >> 12)
302 {
303 case 0x0: /* 0000 - 1fff */
304 case 0x1:
305 ramEnabled = ((val & 0xf) == 0xa);
306 break;
307 case 0x2: /* 2000 - 3fff */
308 case 0x3:
309 val &= 0x7f;
310 refreshRomBank((val) ? val : 1);
311 break;
312 case 0x4: /* 4000 - 5fff */
313 case 0x5:
314 /* The RTC register is selected by writing values 0x8-0xc, ram banks
315 * are selected by values 0x0-0x3 */
316 if (val <= 0x3)
317 refreshRamBank(val);
318 else if (val >= 8 && val <= 0xc)
319 currentRamBank = val;
320 break;
321 case 0x6: /* 6000 - 7fff */
322 case 0x7:
323 if (val)
324 latchClock();
325 break;
326 case 0xa: /* a000 - bfff */
327 case 0xb:
328 if (!ramEnabled)
329 break;
330
331 switch (currentRamBank)
332 { // Check for RTC register
333 case 0x8:
334 if (gbClock.mbc3.s != val)
335 {
336 gbClock.mbc3.s = val;
337 writeClockStruct();
338 }
339 return;
340 case 0x9:
341 if (gbClock.mbc3.m != val)
342 {
343 gbClock.mbc3.m = val;
344 writeClockStruct();
345 }
346 return;
347 case 0xA:
348 if (gbClock.mbc3.h != val)
349 {
350 gbClock.mbc3.h = val;
351 writeClockStruct();
352 }
353 return;
354 case 0xB:
355 if ((gbClock.mbc3.d & 0xff) != val)
356 {
357 gbClock.mbc3.d &= 0x100;
358 gbClock.mbc3.d |= val;
359 writeClockStruct();
360 }
361 return;
362 case 0xC:
363 if (gbClock.mbc3.ctrl != val)
364 {
365 gbClock.mbc3.d &= 0xFF;
366 gbClock.mbc3.d |= (val & 1) << 8;
367 gbClock.mbc3.ctrl = val;
368 writeClockStruct();
369 }
370 return;
371 default: // Not an RTC register
372 if (numRamBanks)
373 writeSram(addr & 0x1fff, val);
374 }
375 break;
376 }

646void latchClock()
647{
648 // +2h, the same as lameboy
649 time_t now = rawTime - 120 * 60;
650 time_t difference = now - gbClock.last;
651 struct tm *lt = gmtime((const time_t *)&difference);
652
653 switch (MBC)
654 {
655 case MBC3:
656 gbClock.mbc3.s += lt->tm_sec;
657 OVERFLOW(gbClock.mbc3.s, 60, gbClock.mbc3.m);
658 gbClock.mbc3.m += lt->tm_min;
659 OVERFLOW(gbClock.mbc3.m, 60, gbClock.mbc3.h);
660 gbClock.mbc3.h += lt->tm_hour;
661 OVERFLOW(gbClock.mbc3.h, 24, gbClock.mbc3.d);
662 gbClock.mbc3.d += lt->tm_yday;
663 /* Overflow! */
664 if (gbClock.mbc3.d > 0x1FF)
665 {
666 /* Set the carry bit */
667 gbClock.mbc3.ctrl |= 0x80;
668 gbClock.mbc3.d -= 0x200;
669 }
670 /* The 9th bit of the day register is in the control register */
671 gbClock.mbc3.ctrl &= ~1;
672 gbClock.mbc3.ctrl |= (gbClock.mbc3.d > 0xff);
673 break;
674 case HUC3:
675 gbClock.huc3.m += lt->tm_min;
676 OVERFLOW(gbClock.huc3.m, 60 * 24, gbClock.huc3.d);
677 gbClock.huc3.d += lt->tm_yday;
678 OVERFLOW(gbClock.huc3.d, 365, gbClock.huc3.y);
679 gbClock.huc3.y += lt->tm_year - 70;
680 break;
681 }
682
683 gbClock.last = now;
684}
