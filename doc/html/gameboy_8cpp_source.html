329void runEmul()
330{
331 for (;;)
332 {
333 cyclesToEvent -= extraCycles;
334 int cycles;
335 if (halt)
336 cycles = cyclesToEvent;
337 else
338 cycles = runOpcode(cyclesToEvent);
339

341 cyclesToExecute = -1;
342

344
345 cyclesToEvent = maxWaitCycles;
346 extraCycles = 0;
347
348 cyclesSinceVblank += cycles;
349
350 if (serialCounter > 0)
351 {
352 serialCounter -= cycles;
353 if (serialCounter <= 0)
354 {
355 serialCounter = 0;
356 linkReceivedData = 0xff;
357 transferReady = true;
358 }
359 else
360 setEventCycles(serialCounter);
361 }
362 if (transferReady)
363 {
364 if (nifiEnabled)
365 {
366 if (!(ioRam[0x02] & 1))
367 {
368 sendPacketByte(56, linkSendData);
369 timerStop(2);
370 }
371 }
372 else if (printerEnabled)

374 sendGbPrinterByte(ioRam[0x01]);
375 }
376 else
377 linkReceivedData = 0xff;
378 ioRam[0x01] = linkReceivedData;
379 requestInterrupt(SERIAL);
380 ioRam[0x02] &= ~0x80;
381 linkReceivedData = -1;
382 transferReady = false;
383 }
384
385 updateTimers(cycles);
386
387 soundCycles += cycles >> doubleSpeed;
388 if (soundCycles >= cyclesToSoundEvent)
389 {
390 cyclesToSoundEvent = 10000;
391 updateSound(soundCycles);
392 soundCycles = 0;
393 }
394 setEventCycles(cyclesToSoundEvent);
395
396 updateLCD(cycles);

398 int interruptTriggered = ioRam[0x0F] & ioRam[0xFF];
399 if (interruptTriggered)
400 {
401 /* Hack to fix Robocop 2 and LEGO Racers, possibly others.
402 * Interrupts can occur in the middle of an opcode. The result of
403 * this is that said opcode can read the resulting state - most
404 * importantly, it can read LY=144 before the vblank interrupt takes
405 * over. This is a decent approximation of that effect.
406 * This has been known to break Megaman V boss intros, that's fixed
407 * by the "opTriggeredInterrupt" stuff.
408 */
409 if (!halt && !opTriggeredInterrupt)
410 extraCycles += runOpcode(4);
411
412 extraCycles += handleInterrupts(interruptTriggered);
413 }

484inline void updateLCD(int cycles)
485{
486 if (!(ioRam[0x40] & 0x80)) // If LCD is off
487 {
488 scanlineCounter = 456 * (doubleSpeed ? 2 : 1);
489 ioRam[0x44] = 0;
490 ioRam[0x41] &= 0xF8;
491 // Normally timing is synchronized with gameboy's vblank. If the screen
492 // is off, this code kicks in. The "phaseCounter" is a counter until the
493 // ds should check for input and whatnot.
494 phaseCounter -= cycles;
495 if (phaseCounter <= 0)
496 {
497 fps++;
498 phaseCounter += 456 * 153 * (doubleSpeed ? 2 : 1);
499 cyclesSinceVblank = 0;
500 // Though not technically vblank, this is a good time to check for
501 // input and whatnot.
502 gameboyUpdateVBlank();
503 }
504 return;
505 }
506
507 scanlineCounter -= cycles;
508
509 if (scanlineCounter > 0)
510 {
511 setEventCycles(scanlineCounter);
512 return;
513 }
514
515 switch (ioRam[0x41] & 3)
516 {
517 case 2:
518 {
519 ioRam[0x41]++; // Set mode 3
520 scanlineCounter += 172 << doubleSpeed;
521 drawScanline(ioRam[0x44]);
522 }
523 break;
524 case 3:
525 {
526 ioRam[0x41] &= ~3; // Set mode 0
527
528 if (ioRam[0x41] & 0x8)
529 requestInterrupt(LCD);
530
531 scanlineCounter += 204 << doubleSpeed;
532
533 drawScanline_P2(ioRam[0x44]);
534 if (updateHblankDMA())
535 {
536 extraCycles += 50;
537 }
538 }
539 break;
540 case 0:
541 {
542 // fall through to next case
543 }
544 case 1:
545 if (ioRam[0x44] == 0 && (ioRam[0x41] & 3) == 1)
546 { // End of vblank
547 ioRam[0x41]++; // Set mode 2
548 scanlineCounter += 80 << doubleSpeed;
549 }
550 else
551 {
552 ioRam[0x44]++;
553
554 if (ioRam[0x44] < 144 || ioRam[0x44] >= 153)
555 { // Not in vblank
556 if (ioRam[0x41] & 0x20)
557 {
558 requestInterrupt(LCD);
559 }
560
561 if (ioRam[0x44] >= 153)
562 {
563 // Don't change the mode. Scanline 0 is twice as
564 // long as normal - half of it identifies as being
565 // in the vblank period.
566 ioRam[0x44] = 0;
567 scanlineCounter += 456 << doubleSpeed;
568 }
569 else
570 { // End of hblank
571 ioRam[0x41] &= ~3;
572 ioRam[0x41] |= 2; // Set mode 2
573 if (ioRam[0x41] & 0x20)
574 requestInterrupt(LCD);
575 scanlineCounter += 80 << doubleSpeed;
576 }
577 }
578
579 checkLYC();
580
581 if (ioRam[0x44] >= 144)
582 { // In vblank
583 scanlineCounter += 456 << doubleSpeed;
584
585 if (ioRam[0x44] == 144) // Beginning of vblank
586 {
587 ioRam[0x41] &= ~3;
588 ioRam[0x41] |= 1; // Set mode 1
589
590 requestInterrupt(VBLANK);
591 if (ioRam[0x41] & 0x10)
592 requestInterrupt(LCD);
593
594 fps++;
595 cyclesSinceVblank = scanlineCounter - (456 << doubleSpeed);
596 gameboyUpdateVBlank();
597 }
598 }
599 }
600
601 break;
602 }
603
604 setEventCycles(scanlineCounter);
605}

607inline void updateTimers(int cycles)
608{
609 if (ioRam[0x07] & 0x4) // Timers enabled
610 {
611 timerCounter -= cycles;
612 while (timerCounter <= 0)
613 {
614 int clocksAdded = (-timerCounter) / timerPeriod + 1;
615 timerCounter += timerPeriod * clocksAdded;
616 int sum = ioRam[0x05] + clocksAdded;
617 if (sum > 0xff)
618 {
619 requestInterrupt(TIMER);
620 ioRam[0x05] = ioRam[0x06];
621 }
622 else
623 ioRam[0x05] = (u8)sum;
624 }
625 // Set cycles until the timer will trigger an interrupt.
626 // Reads from [0xff05] may be inaccurate.
627 // However Castlevania and Alone in the Dark are extremely slow
628 // if this is updated each time [0xff05] is changed.
629 setEventCycles(timerCounter + timerPeriod * (255 - ioRam[0x05]));
630 }
631 dividerCounter -= cycles;
632 if (dividerCounter <= 0)
633 {
634 int divsAdded = -dividerCounter / 256 + 1;
635 dividerCounter += divsAdded * 256;
636 ioRam[0x04] += divsAdded;
637 }
638 // setEventCycles(dividerCounter);
639}
