728int loadSave()
729{
730 if (saveFile != NULL)
731 {
732 fclose(saveFile);
733 saveFile = NULL;
734 }
735 if (externRam != NULL)
736 {
737 free(externRam);
738 externRam = NULL;
739 }
740
741 if (gbsMode)
742 numRamBanks = 1;
743 else
744 {
745 // Get the game's external memory size and allocate the memory
746 switch (ramSize)
747 {
748 case 0:
749 numRamBanks = 0;
750 break;
751 case 1:
752 case 2:
753 numRamBanks = 1;
754 break;
755 case 3:
756 numRamBanks = 4;
757 break;
758 case 4:
759 numRamBanks = 16;
760 break;
761 default:
762 printLog("Invalid RAM bank number: %x\nDefaulting to 4 banks\n", ramSize);
763 numRamBanks = 4;
764 break;
765 }
766 if (MBC == MBC2)
767 numRamBanks = 1;
768 }
769
770 if (numRamBanks == 0)
771 return 0;
772
773 externRam = (u8 *)malloc(numRamBanks * 0x2000);
774
775 if (gbsMode)
776 return 0; // GBS files don't get to save.
777
778 // Now load the data.
779 saveFile = fopen(savename, "r+b");
780 int neededFileSize = numRamBanks * 0x2000;
781 if (MBC == MBC3 || MBC == HUC3)
782 neededFileSize += sizeof(clockStruct);
783
784 int fileSize = 0;
785 if (saveFile)
786 {
787 fseek(saveFile, 0, SEEK_END);
788 fileSize = ftell(saveFile);
789 fseek(saveFile, 0, SEEK_SET);
790 }
791
792 if (!saveFile || fileSize < neededFileSize)
793 {
794 fclose(saveFile);
795
796 // Extend the size of the file, or create it
797 if (!saveFile)
798 {
799 saveFile = fopen(savename, "wb");
800 fseek(saveFile, neededFileSize - 1, SEEK_SET);
801 fputc(0, saveFile);
802 }
803 else
804 {
805 saveFile = fopen(savename, "ab");
806 for (; fileSize < neededFileSize; fileSize++)
807 fputc(0, saveFile);
808 }
809 fclose(saveFile);
810
811 saveFile = fopen(savename, "r+b");
812 }
813
814 fread(externRam, 1, 0x2000 * numRamBanks, saveFile);
815
816 switch (MBC)
817 {
818 case MBC3:
819 case HUC3:
820 fread(&gbClock, 1, sizeof(gbClock), saveFile);
821 break;
822 }
823 saveGame();
824 return 0;

855{
