54int nameSortFunction(char *&a, char *&b)
55{
56 // ".." sorts before everything except itself.
57 bool aIsParent = strcmp(a, "..") == 0;
58 bool bIsParent = strcmp(b, "..") == 0;
59
60 if (aIsParent && bIsParent)
61 return 0;
62 else if (aIsParent) // Sorts before
63 return -1;
64 else if (bIsParent) // Sorts after
65 return 1;
66 else
67 return strcasecmp(a, b);
68}

70/*
71 * Determines whether a portion of a vector is sorted.
72 * Input assertions: 'from' and 'to' are valid indices into data. 'to' can be
73 * the maximum value for the type 'unsigned int'.
74 * Input: 'data', data vector, possibly sorted.
75 * 'sortFunction', function determining the sort order of two elements.
76 * 'from', index of the first element in the range to test.
77 * 'to', index of the last element in the range to test.
78 * Output: true if, for any valid index 'i' such as from <= i < to,
79 * data[i] < data[i + 1].
80 * true if the range is one or no elements, or if from > to.
81 * false otherwise.
82 */
83template <class Data>
84bool isSorted(std::vector<Data> &data, int (*sortFunction)(Data &, Data &), const unsigned int from, const unsigned int to)
85{
86 if (from >= to)
87 return true;
88
89 Data *prev = &data[from];
90 for (unsigned int i = from + 1; i < to; i++)
91 {
92 if ((*sortFunction)(*prev, data[i]) > 0)
93 return false;
94 prev = &data[i];
95 }
96 if ((*sortFunction)(*prev, data[to]) > 0)
97 return false;
98
99 return true;
100}

102/*
103 * Chooses a pivot for Quicksort. Uses the median-of-three search algorithm
104 * first proposed by Robert Sedgewick.
105 * Input assertions: 'from' and 'to' are valid indices into data. 'to' can be
106 * the maximum value for the type 'unsigned int'.
107 * Input: 'data', data vector.
108 * 'sortFunction', function determining the sort order of two elements.
109 * 'from', index of the first element in the range to be sorted.
110 * 'to', index of the last element in the range to be sorted.
111 * Output: a valid index into data, between 'from' and 'to' inclusive.
112 */
113template <class Data>
114unsigned int choosePivot(std::vector<Data> &data, int (*sortFunction)(Data &, Data &), const unsigned int from, const unsigned int to)
115{
116 // The highest of the two extremities is calculated first.
117 unsigned int highest = ((*sortFunction)(data[from], data[to]) > 0)
118 ? from
119 : to;
120 // Then the lowest of that highest extremity and the middle
121 // becomes the pivot.
122 return ((*sortFunction)(data[from + (to - from) / 2], data[highest]) < 0)
123 ? (from + (to - from) / 2)
124 : highest;
125}

127/*
128 * Partition function for Quicksort. Moves elements such that those that are
129 * less than the pivot end up before it in the data vector.
130 * Input assertions: 'from', 'to' and 'pivotIndex' are valid indices into data.
131 * 'to' can be the maximum value for the type 'unsigned int'.
132 * Input: 'data', data vector.
133 * 'metadata', data describing the values in 'data'.
134 * 'sortFunction', function determining the sort order of two elements.
135 * 'from', index of the first element in the range to sort.
136 * 'to', index of the last element in the range to sort.
137 * 'pivotIndex', index of the value chosen as the pivot.
138 * Output: the index of the value chosen as the pivot after it has been moved
139 * after all the values that are less than it.
140 */
141template <class Data, class Metadata>
142unsigned int partition(std::vector<Data> &data, std::vector<Metadata> &metadata, int (*sortFunction)(Data &, Data &), const unsigned int from, const unsigned int to, const unsigned int pivotIndex)
143{
144 Data pivotValue = data[pivotIndex];
145 data[pivotIndex] = data[to];
146 data[to] = pivotValue;
147 {
148 const Metadata tM = metadata[pivotIndex];
149 metadata[pivotIndex] = metadata[to];
150 metadata[to] = tM;
151 }
152
153 unsigned int storeIndex = from;
154 for (unsigned int i = from; i < to; i++)
155 {
156 if ((*sortFunction)(data[i], pivotValue) < 0)
157 {
158 const Data tD = data[storeIndex];
159 data[storeIndex] = data[i];
160 data[i] = tD;
161 const Metadata tM = metadata[storeIndex];
162 metadata[storeIndex] = metadata[i];
163 metadata[i] = tM;
164 ++storeIndex;
165 }
166 }
167
168 {
169 const Data tD = data[to];
170 data[to] = data[storeIndex];
171 data[storeIndex] = tD;
172 const Metadata tM = metadata[to];
173 metadata[to] = metadata[storeIndex];
174 metadata[storeIndex] = tM;
175 }
176 return storeIndex;
177}

179/*
180 * Sorts an array while keeping metadata in sync.
181 * This sort is unstable and its average performance is
182 * O(data.size() * log2(data.size()).
183 * Input assertions: for any valid index 'i' in data, index 'i' is valid in
184 * metadata. 'from' and 'to' are valid indices into data. 'to' can be
185 * the maximum value for the type 'unsigned int'.
186 * Invariant: index 'i' in metadata describes index 'i' in data.
187 * Input: 'data', data to sort.
188 * 'metadata', data describing the values in 'data'.
189 * 'sortFunction', function determining the sort order of two elements.
190 * 'from', index of the first element in the range to sort.
191 * 'to', index of the last element in the range to sort.
192 */
193template <class Data, class Metadata>
194void quickSort(std::vector<Data> &data, std::vector<Metadata> &metadata, int (*sortFunction)(Data &, Data &), const unsigned int from, const unsigned int to)
195{
196 if (isSorted(data, sortFunction, from, to))
197 return;
198
199 unsigned int pivotIndex = choosePivot(data, sortFunction, from, to);
200 unsigned int newPivotIndex = partition(data, metadata, sortFunction, from, to, pivotIndex);
201 if (newPivotIndex > 0)
202 quickSort(data, metadata, sortFunction, from, newPivotIndex - 1);
203 if (newPivotIndex < to)
204 quickSort(data, metadata, sortFunction, newPivotIndex + 1, to);
205}
