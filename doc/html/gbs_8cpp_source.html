61void gbsLoadSong()
62{
63 initMMU();
64 ime = 0;
65
66 gbRegs.sp.w = READ16(gbsHeader + 0x0c);
67 u8 tma = gbsHeader[0x0e];
68 u8 tac = gbsHeader[0x0f];
69
70 if (tac & 0x80)
71 setDoubleSpeed(1);
72 tac &= ~0x80;
73 if (tma == 0 && tac == 0)
74 {
75 // Vblank interrupt handler
76 romSlot0[0x40] = 0xcd; // call
77 romSlot0[0x41] = gbsPlayAddress & 0xff;
78 romSlot0[0x42] = gbsPlayAddress >> 8;
79 romSlot0[0x43] = 0xd9; // reti
80 writeIO(0xff, VBLANK);
81 }
82 else
83 {
84 // Timer interrupt handler
85 romSlot0[0x50] = 0xcd; // call
86 romSlot0[0x51] = gbsPlayAddress & 0xff;
87 romSlot0[0x52] = gbsPlayAddress >> 8;
88 romSlot0[0x53] = 0xd9; // reti
89 writeIO(0xff, TIMER);
90 }
91
92 writeIO(0x05, 0x00);
93 writeIO(0x06, tma);
94 writeIO(0x07, tac);
95
96 gbsPlayingSong = gbsSelectedSong;
97 gbRegs.af.b.h = gbsPlayingSong;
98 writeMemory(--gbRegs.sp.w, 0x01);
99 writeMemory(--gbRegs.sp.w, 0x00); // Will return to beginning
