81void doVramTransfer(u8 *dest)
82{
83 // Set the base address of the background map based on the value in IO register 0x40.
84 int map = 0x1800 + ((ioRam[0x40] >> 3) & 1) * 0x400;
85 int index = 0;
86
87 // Iterate through each tile in the background map and copy its data to the destination buffer.
88 for (int y = 0; y < 18; y++)
89 {
90 for (int x = 0; x < 20; x++)
91 {
92 // Stop copying data if we reach the end of the destination buffer.
93 if (index == 0x1000)
94 return;
95
96 // Get the index of the tile in the VRAM based on its position in the background map.
97 int tile = vram[0][map + y * 32 + x];
98
99 // Copy the tile data to the destination buffer.
100 if (ioRam[0x40] & 0x10)
101 memcpy(dest + index, vram[0] + tile * 16, 16);
102 else
103 memcpy(dest + index, vram[0] + 0x1000 + ((s8)tile) * 16, 16);
104
105 // Increment the index to the next chunk of data in the destination buffer.
106 index += 16;
107 }
108 }

627void sgbHandleP1(u8 val)
628{
629 if ((val & 0x30) == 0)
630 {
631 // Start packet transfer
632 sgbPacketBit = 0;
633 ioRam[0x00] = 0xcf;
634 return;
635 }
636
637 if (sgbPacketBit != -1)
638 {
639 // Continue packet transfer
640 u8 oldVal = ioRam[0x00];
641 ioRam[0x00] = val;
642
643 int shift = sgbPacketBit % 8;
644 int byte = (sgbPacketBit / 8) % 16;
645
646 // If the current bit is the first bit of a byte, clear it first.
647 if (shift == 0)
648 sgbPacket[byte] = 0;
649
650 int bit;
651 if ((oldVal & 0x30) == 0 && (val & 0x30) != 0x30)
652 { // A bit of speculation here. Fixes Castlevania.
653 // If a packet transfer is interrupted, stop it.
654 sgbPacketBit = -1;
655 return;
656 }
657 if (!(val & 0x10))
658 bit = 0;
659 else if (!(val & 0x20))
660 bit = 1;
661 else
662 return;
663
664 // Add the current bit to the packet.
665 sgbPacket[byte] |= bit << shift;
666 sgbPacketBit++;
667
668 // If the packet is complete, execute the corresponding command.
669 if (sgbPacketBit == 128)
670 {
671 if (sgbPacketsTransferred == 0)
672 {
673 sgbCommand = sgbPacket[0] / 8;
674 sgbPacketLength = sgbPacket[0] & 7;
675 // printLog("CMD %x\n", sgbCommand);
676 }
677 if (sgbCommands[sgbCommand] != 0)
678 sgbCommands[sgbCommand](sgbPacketsTransferred);
679
680 sgbPacketBit = -1;
681 sgbPacketsTransferred++;
682
683 // If all packets for a command have been transferred, reset the counters.
684 if (sgbPacketsTransferred == sgbPacketLength)
685 {
686 sgbPacketLength = 0;
687 sgbPacketsTransferred = 0;
688 }
689 }
690 }
691 else
692 {
693 // Handle controller input
694 if ((val & 0x30) == 0x30)
695 {
696 if (buttonsChecked == 3)
697 {
698 selectedController++;
699 if (selectedController >= numControllers)
700 selectedController = 0;
701 buttonsChecked = 0;
702 }
703 ioRam[0x00] = 0xff - selectedController;
704 }
705 else
706 {
707 ioRam[0x00] = val | 0xcf;
708 if ((val & 0x30) == 0x10)
709 buttonsChecked |= 1;
710 else if ((val & 0x30) == 0x20)
711 buttonsChecked |= 2;
712 }
713 }
