30void fifoValue32Handler(u32 value, void *user_data)
31{
32 static u8 scalingWasOn;
33 static bool soundWasDisabled;
34 static bool wasPaused;
35 switch (value)
36 {
37 case FIFOMSG_LID_CLOSED:
38 // Entering sleep mode
39 scalingWasOn = sharedData->scalingOn;
40 soundWasDisabled = soundDisabled;
41 wasPaused = isGameboyPaused();
42
43 sharedData->scalingOn = 0;
44 soundDisabled = true;
45 pauseGameboy();
46 break;
47 case FIFOMSG_LID_OPENED:
48 // Exiting sleep mode
49 sharedData->scalingOn = scalingWasOn;
50 soundDisabled = soundWasDisabled;
51 if (!wasPaused)
52 unpauseGameboy();
53 // Time isn't incremented properly in sleep mode, compensate here.
54 time(&rawTime);
55 lastRawTime = rawTime;
56 break;
57 }
58}
