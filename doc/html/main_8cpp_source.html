41void fifoValue32Handler(u32 value, void *user_data)
42{
43 static u8 scalingWasOn;
44 static bool soundWasDisabled;
45 static bool wasPaused;
47 switch (value)
48 {
49 case FIFOMSG_LID_CLOSED:
50 // Entering sleep mode
51 scalingWasOn = sharedData->scalingOn;
52 soundWasDisabled = soundDisabled;
53 wasPaused = isGameboyPaused();
54
55 sharedData->scalingOn = 0;
56 soundDisabled = true;
57 pauseGameboy();
58 break;
59 case FIFOMSG_LID_OPENED:
60 // Exiting sleep mode
61 sharedData->scalingOn = scalingWasOn;
62 soundDisabled = soundWasDisabled;
63 if (!wasPaused)
64 unpauseGameboy();
65
66 // Time isn't incremented properly in sleep mode, compensate here.
67 time(&rawTime);
68 lastRawTime = rawTime;
69 break;
70 }
71}

163void initializeGameboy()
164{
165 enableSleepMode(); // Set the Gameboy to sleep mode
166 gameboyFrameCounter = 0; // Reset the frame counter to 0
167 sgbMode = false; // Set the Super Gameboy mode to false
168
169 if (gbsMode) // Check if it is in Gameboy Sound mode
170 {
171 resultantGBMode = 1; // Set the resultant Gameboy mode to Gameboy Color
172 probingForBorder = false; // Stop probing for border
173 }
174 else // If not in Gameboy Sound mode
175 {
176 switch (gbcModeOption) // Check the Gameboy Color mode option
177 {
178 case 0: // Gameboy mode
179 initGBMode(); // Initialize Gameboy mode
180 break;
181 case 1: // Gameboy Color mode if needed
182 if (romSlot0[0x143] == 0xC0)
183 initGBCMode(); // Initialize Gameboy Color mode
184 else
185 initGBMode(); // Initialize Gameboy mode
186 break;
187 case 2: // Gameboy Color mode
188 if (romSlot0[0x143] == 0x80 || romSlot0[0x143] == 0xC0)
189 initGBCMode(); // Initialize Gameboy Color mode
190 else
191 initGBMode(); // Initialize Gameboy mode
192 break;
193 }
194
195 bool sgbEnhanced = romSlot0[0x14b] == 0x33 && romSlot0[0x146] == 0x03; // Check if the ROM is Super Gameboy enhanced
196 if (sgbEnhanced && resultantGBMode != 2 && probingForBorder)
197 {
198 resultantGBMode = 2; // Set the resultant Gameboy mode to Super Gameboy
199 }
200 else
201 {
202 probingForBorder = false; // Stop probing for border
203 }
204 } // !gbsMode
205
206 initMMU(); // Initialize the Memory Management Unit
207 initCPU(); // Initialize the Central Processing Unit
208 initLCD(); // Initialize the Liquid Crystal Display
209 initGFX(); // Initialize the Graphics
210 initSND(); // Initialize the Sound
211
212 if (!gbsMode && !probingForBorder && suspendStateExists)
213 {
214 loadState(-1); // Load the suspended state
215 }
216
217 if (gbsMode)
218 gbsInit(); // Initialize the Gameboy Sound mode
219
220 // We haven't calculated the # of cycles to the next hardware event.
221 cyclesToEvent = 1; // Set the number of cycles to the next hardware event to 1
222}
