521void redrawMenu()
522{
523 PrintConsole *oldConsole = getPrintConsole();
524 setPrintConsole(menuConsole);
525 consoleClear();
526
527 // Top line: submenu
528 int pos = 0;
529 int nameStart = (32 - strlen(menuList[menu].name) - 2) / 2;
530 if (option == -1)
531 {
532 nameStart -= 2;
533 iprintfColored(CONSOLE_COLOR_LIGHT_GREEN, "<");
534 }
535 else
536 iprintf("<");
537 pos++;
538 for (; pos < nameStart; pos++)
539 iprintf(" ");
540 if (option == -1)
541 {
542 iprintfColored(CONSOLE_COLOR_LIGHT_YELLOW, "* ");
543 pos += 2;
544 }
545 {
546 int color = (option == -1 ? CONSOLE_COLOR_LIGHT_YELLOW : CONSOLE_COLOR_WHITE);
547 iprintfColored(color, "[%s]", menuList[menu].name);
548 }
549 pos += 2 + strlen(menuList[menu].name);
550 if (option == -1)
551 {
552 iprintfColored(CONSOLE_COLOR_LIGHT_YELLOW, " *");
553 pos += 2;
554 }
555 for (; pos < 31; pos++)
556 iprintf(" ");
557 if (option == -1)
558 iprintfColored(CONSOLE_COLOR_LIGHT_GREEN, ">");
559 else
560 iprintf(">");
561 iprintf("\n");
562
563 // Rest of the lines: options
564 for (int i = 0; i < menuList[menu].numOptions; i++)
565 {
566 int option_color;
567 if (!menuList[menu].options[i].enabled)
568 option_color = CONSOLE_COLOR_GREY;
569 else if (option == i)
570 option_color = CONSOLE_COLOR_LIGHT_YELLOW;
571 else
572 option_color = CONSOLE_COLOR_WHITE;
573
574 if (menuList[menu].options[i].numValues == 0)
575 {
576 for (unsigned int j = 0; j < (32 - strlen(menuList[menu].options[i].name)) / 2 - 2; j++)
577 iprintf(" ");
578 if (i == option)
579 {
580 iprintfColored(option_color, "* %s *\n\n", menuList[menu].options[i].name);
581 }
582 else
583 iprintfColored(option_color, " %s \n\n", menuList[menu].options[i].name);
584 }
585 else
586 {
587 for (unsigned int j = 0; j < 16 - strlen(menuList[menu].options[i].name); j++)
588 iprintf(" ");
589 if (i == option)
590 {
591 iprintfColored(option_color, "* ");
592 iprintfColored(option_color, "%s ", menuList[menu].options[i].name);
593 iprintfColored(menuList[menu].options[i].enabled ? CONSOLE_COLOR_LIGHT_GREEN : option_color,
594 "%s", menuList[menu].options[i].values[menuList[menu].options[i].selection]);
595 iprintfColored(option_color, " *");
596 }
597 else
598 {
599 iprintf(" ");
600 iprintfColored(option_color, "%s ", menuList[menu].options[i].name);
601 iprintfColored(option_color, "%s", menuList[menu].options[i].values[menuList[menu].options[i].selection]);
602 }
603 iprintf("\n\n");
604 }
605 }
606
607 // Message at the bottom
608 if (printMessage[0] != '\0')
609 {
610 int newlines = 23 - (menuList[menu].numOptions * 2 + 2) - 1;
611 for (int i = 0; i < newlines; i++)
612 iprintf("\n");
613 int spaces = 31 - strlen(printMessage);
614 for (int i = 0; i < spaces; i++)
615 iprintf(" ");
616 iprintf("%s\n", printMessage);
617
618 printMessage[0] = '\0';
619 }
620
621 setPrintConsole(oldConsole);
622}

624// Called each vblank while the menu is on
625void updateMenu()
626{
627 if (subMenuUpdateFunc != 0)
628 {
629 subMenuUpdateFunc();
630 return;
631 }
632
633 bool redraw = false;
634 // Get input
635 if (keyPressedAutoRepeat(KEY_UP))
636 {
637 option--;
638 if (option < -1)
639 option = menuList[menu].numOptions - 1;
640 redraw = true;
641 }
642 else if (keyPressedAutoRepeat(KEY_DOWN))
643 {
644 option++;
645 if (option >= menuList[menu].numOptions)
646 option = -1;
647 redraw = true;
648 }
649 else if (keyPressedAutoRepeat(KEY_LEFT))
650 {
651 if (option == -1)
652 {
653 menu--;
654 if (menu < 0)
655 menu = numMenus - 1;
656 }
657 else if (menuList[menu].options[option].numValues != 0 && menuList[menu].options[option].enabled)
658 {
659 int selection = menuList[menu].options[option].selection - 1;
660 if (selection < 0)
661 selection = menuList[menu].options[option].numValues - 1;
662 menuList[menu].options[option].selection = selection;
663 menuList[menu].options[option].function(selection);
664 }
665 redraw = true;
666 }
667 else if (keyPressedAutoRepeat(KEY_RIGHT))
668 {
669 if (option == -1)
670 {
671 menu++;
672 if (menu >= numMenus)
673 menu = 0;
674 }
675 else if (menuList[menu].options[option].numValues != 0 && menuList[menu].options[option].enabled)
676 {
677 int selection = menuList[menu].options[option].selection + 1;
678 if (selection >= menuList[menu].options[option].numValues)
679 selection = 0;
680 menuList[menu].options[option].selection = selection;
681 menuList[menu].options[option].function(selection);
682 }
683 redraw = true;
684 }
685 else if (keyJustPressed(KEY_A))
686 {
687 forceReleaseKey(KEY_A);
688 if (option >= 0 && menuList[menu].options[option].numValues == 0 && menuList[menu].options[option].enabled)
689 {
690 menuList[menu].options[option].function(menuList[menu].options[option].selection);
691 }
692 redraw = true;
693 }
694 else if (keyJustPressed(KEY_B))
695 {
696 forceReleaseKey(KEY_B);
697 closeMenu();
698 updateScreens();
699 }
700 else if (keyJustPressed(KEY_L))
701 {
702 menu--;
703 if (menu < 0)
704 menu = numMenus - 1;
705 if (option >= menuList[menu].numOptions)
706 option = menuList[menu].numOptions - 1;
707 redraw = true;
708 }
709 else if (keyJustPressed(KEY_R))
710 {
711 menu++;
712 if (menu >= numMenus)
713 menu = 0;
714 if (option >= menuList[menu].numOptions)
715 option = menuList[menu].numOptions - 1;
716 redraw = true;
717 }
718 if (redraw && subMenuUpdateFunc == 0 &&
719 isMenuOn()) // The menu may have been closed by an option
720 doAtVBlank(redrawMenu);
721}
