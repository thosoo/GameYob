42void doCommand(u32 command)
43{
44 int cmd = (command >> 20) & 0xf;
45 int data = command & 0xFFFF;
46 int i;
47
48 switch (cmd)
49 {
50
51 case GBSND_UPDATE_COMMAND:
52 if (data == 4)
53 {
54 for (i = 0; i < 4; i++)
55 {
56 updateChannel(i, true);
57 }
58 }
59 else
60 updateChannel(data, true);
61 break;
62
63 case GBSND_START_COMMAND:
64 startChannel(data);
65 break;
66
67 case GBSND_VOLUME_COMMAND:
68 setChannelVolume(data, true);
69 break;
70
71 case GBSND_MASTER_VOLUME_COMMAND:
72 updateMasterVolume();
73 break;
74
75 case GBSND_KILL_COMMAND:
76 // SCHANNEL_CR(channel) &= ~SCHANNEL_ENABLE;
77 break;
78
79 case GBSND_MUTE_COMMAND:
80 // This does not touch the "background hum" to prevent clicking.
81 for (i = 0; i < 4; i++)
82 {
83 SCHANNEL_CR(channels[i]) &= ~0x7f;
84 }
85 break;
86
87 case GBSND_UNMUTE_COMMAND:
88 for (i = 0; i < 4; i++)
89 {
90 if ((sharedData->chanOn & (1 << i)) && sharedData->chanEnabled[i])
91 setChannelVolume(i, true);
92 }
93 break;
94
95 case GBSND_HYPERSOUND_ENABLE_COMMAND:
96 setHyperSound(data);
97 break;
98
99 default:
100 return;
101 }
102}

150void startChannel(int c)
151{
152 int channel = channels[c];
153
154 if (!sharedData->chanEnabled[c])
155 {

157 return;

159
160 if (c == 2)

163 SCHANNEL_REPEAT_POINT(channel) = 0;
164 SCHANNEL_LENGTH(channel) = 0x20 >> 2;

195void updateChannel(int c, bool write)
196{
197 int channel = channels[c];
198
199 if (!(sharedData->chanOn & (1 << c)) || !sharedData->chanEnabled[c])
200 {
201 schannelCR[c] &= ~0x7f;
202 SCHANNEL_CR(channel) &= ~0x7f; // Set volume to zero
203 return;
204 }
205
206 SCHANNEL_TIMER(channel) = SOUND_FREQ(sharedData->chanRealFreq[c]);
207 if (c < 2)
208 {
209 schannelCR[c] &= ~(SOUND_PAN(127) | 7 << 24);
210 schannelCR[c] |= SOUND_PAN(sharedData->chanPan[c]) | (dutyIndex[sharedData->chanDuty[c]] << 24);
211 }
212 else if (c == 2)
213 {
214 schannelCR[c] &= ~(SOUND_PAN(127));
215 schannelCR[c] |= SOUND_PAN(sharedData->chanPan[c]);
216 }
217 else if (c == 3)
218 {
219 if (currentLfsr != sharedData->lfsr7Bit)
220 startChannel(c);
221 schannelCR[c] &= ~(SOUND_PAN(127));
222 schannelCR[c] |= SOUND_PAN(sharedData->chanPan[c]);
223 }
224 if (write)
225 {
226 SCHANNEL_CR(channel) &= ~(SOUND_PAN(127) | 7 << 24);
227 SCHANNEL_CR(channel) |= (schannelCR[c] & (SOUND_PAN(127) | 7 << 24));
228 }
229 setChannelVolume(c, write);
230}

242 // Each sound channel enabled in NR51 adds a bit of a "background tone".
243 // I'm not really sure if this behaviour is correct.
244 // I'm trying to strike a balance between the "Warlocked/Perfect Dark/Alone
245 // in the Dark" sound effects, and the annoying clicking in some games.
246 int vol = 0;
